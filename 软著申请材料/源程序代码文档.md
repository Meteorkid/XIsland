  97      /// Parse test result patterns from tool output (e.g. pytest / jest / go test summary lines).
  98      static func parseTestResults(from output: String?) -> TestResults? {
  99          guard let output = output else { return nil }
 100  
 101          // Python / pytest: allow missing "failed" when zero failures
 102          if let passed = firstInt(matching: #"(\d+)\s*passed"#, in: output) {
 103              let failed = firstInt(matching: #"(\d+)\s*failed"#, in: output) ?? 0
 104              let skipped = firstInt(matching: #"(\d+)\s*skipped"#, in: output) ?? 0
 105              let total = passed + failed + skipped
 106              if total > 0 { return TestResults(passed: passed, failed: failed, skipped: skipped, total: total) }
 107          }
 108  
 109          // jest / vitest: prefer explicit failed + skipped when present
 110          if output.localizedCaseInsensitiveContains("tests:"),
 111             let passed = firstInt(matching: #"(\d+)\s+passed"#, in: output),
 112             let total = firstInt(matching: #"(\d+)\s+total"#, in: output) {
 113              let failed = firstInt(matching: #"(\d+)\s+failed"#, in: output)
 114                  ?? max(0, total - passed - (firstInt(matching: #"(\d+)\s+skipped"#, in: output) ?? 0))
 115              let skipped = firstInt(matching: #"(\d+)\s+skipped"#, in: output) ?? 0
 116              return TestResults(passed: passed, failed: failed, skipped: skipped, total: total)
 117          }
 118  
 119          // go test package summary: `ok   pkg 0.1s` or `FAIL\tpkg`
 120          if let match = firstMatch(
 121              pattern: #"^(ok|FAIL)\s+(\S+)"#,
 122              options: [.anchorsMatchLines],
 123              in: output
 124          ), !match.isEmpty {
 125              let statusToken = match[0].uppercased()
 126              let passed = statusToken == "OK" ? 1 : 0
 127              let failed = statusToken == "FAIL" ? 1 : 0
 128              return TestResults(passed: passed, failed: failed, skipped: 0, total: 1)
 129          }
 130  
 131          // Legacy: "--- PASS:" / "--- FAIL:" line counts
 132          let passCount = countOccurrences(of: "--- PASS:", in: output)
 133          let failCount = countOccurrences(of: "--- FAIL:", in: output)
 134          if passCount > 0 || failCount > 0 {
 135              return TestResults(passed: passCount, failed: failCount, skipped: 0, total: passCount + failCount)
 136          }
 137  
 138          return nil
 139      }

 143          guard let output = output, !output.isEmpty else { return nil }

 151      private static func firstMatch(pattern: String, options: NSRegularExpression.Options, in text: String) -> [String]? {
 152          guard let regex = try? NSRegularExpression(pattern: pattern, options: options) else { return nil }
 153          let range = NSRange(text.startIndex..<text.endIndex, in: text)
 154          guard let match = regex.firstMatch(in: text, options: [], range: range) else { return nil }
 155          var caps: [String] = []
 156          for i in 1..<match.numberOfRanges {
 157              guard let r = Range(match.range(at: i), in: text) else { return nil }
 158              caps.append(String(text[r]))
 159          }
 160          return caps
 161      }
 162  
 163      private static func firstInt(matching pattern: String, in text: String) -> Int? {
 164          guard let regex = try? NSRegularExpression(pattern: pattern, options: [.caseInsensitive]) else {
 165              return nil
 166          }
 167          let range = NSRange(text.startIndex..<text.endIndex, in: text)
 168          guard let match = regex.firstMatch(in: text, options: [], range: range),
 169                match.numberOfRanges > 1,
 170                let captureRange = Range(match.range(at: 1), in: text)
 171          else { return nil }
 172          return Int(text[captureRange])
 173      }
 174  
 175      private static func countOccurrences(of substring: String, in text: String) -> Int {
 176          var count = 0
 177          var searchRange = text.startIndex..<text.endIndex
 178          while let foundRange = text.range(of: substring, range: searchRange) {
 179              count += 1
 180              searchRange = foundRange.upperBound..<text.endIndex
 181          }
 182          return count
 183      }

  45          }
  46  
  47          let fd = connectSocket()
  48          guard fd >= 0 else {
  49              if needsJsonOutput(hookType) {
  50                  print("{}")
  51              }
  52              exit(0)
  53          }
  54  
  55          encoded.withUnsafeBytes { ptr in
  56              if let base = ptr.baseAddress {
  57                  _ = send(fd, base, ptr.count, 0)
  58              }
  59          }
  60          shutdown(fd, SHUT_WR)
  61  
  62          if isInteractive {
  63              dumpStdin(hook: "AWAIT_RESPONSE", data: ["type": message.type.rawValue, "fd": "\(fd)"])
  64              let response = receiveResponse(fd)
  65              dumpStdin(hook: "RECV_DONE", data: ["got": response != nil ? response!.type.rawValue : "nil"])
  66              close(fd)
  67  
  68              if let response {
  69                  switch response.type {
  70                  case .permissionResponse:
  71                      let approved = response.approved ?? false
  72                      // Claude Code reads the PermissionRequest outcome from stdout JSON (hookSpecificOutput),
  73                      // not from exit status alone. See hooks.md § PermissionRequest decision control.
  74                      if agentType == "claude_code" {
  75                          print(Self.buildClaudeCodePermissionResponse(approved: approved))
  76                      }
  77                      dumpStdin(hook: "PERM_EXIT", data: ["approved": "\(approved)"])
  78                      exit(approved ? 0 : 1)
  79  
  80                  case .questionResponse:
  81                      let answer = response.answer ?? ""
  82                      if agentType == "cursor" {
  83                          let jsonObj: [String: Any] = [
  84                              "permission": "deny",
  85                              "agent_message": "The user already answered this question via X Island. User selected: \(answer). Do NOT ask the same question again. Continue with the conversation using this answer."
  86                          ]
  87                          if let data = try? JSONSerialization.data(withJSONObject: jsonObj),
  88                             let str = String(data: data, encoding: .utf8) {
  89                              print(str)
  90                          }
  91                      } else if isClaudeCodeQuestion(hookType: hookType, stdinData: stdinData) {
  92                          let json = buildClaudeCodeQuestionResponse(
  93                              answer: answer,
  94                              stdinData: stdinData,
  95                              hookType: hookType
  96                          )
  97                          print(json)
  98                      } else {
  99                          print(answer)
 100                      }
 101                      exit(0)
 102  
 103                  case .planResponse:
 104                      let approved = response.planApproved ?? false
 105                      if let feedback = response.feedback {
 106                          print(feedback)
 107                      }
 108                      exit(approved ? 0 : 1)
 109  
 110                  default:
 111                      exit(0)
 112                  }
 113              } else {
 114                  fputs("[di-bridge] No response received\n", stderr)
 115                  exit(1)
 116              }
 117          } else {
 118              close(fd)
 119              if needsJsonOutput(hookType) {
 120                  print("{}")
 121              }
 122              exit(0)
 123          }
 124      }

 128      static func connectSocket() -> Int32 {
 129          let fd = socket(AF_UNIX, SOCK_STREAM, 0)
 130          guard fd >= 0 else { return -1 }
 131  
 132          var addr = sockaddr_un()
 133          addr.sun_family = sa_family_t(AF_UNIX)
 134          let path = DISocketConfig.socketPath
 135          let pathBytes = path.utf8CString
 136          withUnsafeMutablePointer(to: &addr.sun_path) { sunPath in
 137              pathBytes.withUnsafeBufferPointer { src in
 138                  UnsafeMutableRawPointer(sunPath)
 139                      .copyMemory(from: src.baseAddress!, byteCount: min(src.count, 104))
 140              }
 141          }
 142  
 143          let addrLen = socklen_t(MemoryLayout<sockaddr_un>.size)
 144          let result = withUnsafePointer(to: &addr) { ptr in
 145              ptr.withMemoryRebound(to: sockaddr.self, capacity: 1) { sockPtr in
 146                  connect(fd, sockPtr, addrLen)
 147              }
 148          }
 149  
 150          if result != 0 {
 151              close(fd)
 152              return -1
 153          }
 154          return fd
 155      }

 157      static func receiveResponse(_ fd: Int32) -> DIMessage? {
 158          var data = Data()
 159          let bufSize = 65536
 160          let buf = UnsafeMutablePointer<UInt8>.allocate(capacity: bufSize)
 161          defer { buf.deallocate() }
 162  
 163          let timeout = timeval(tv_sec: 300, tv_usec: 0)
 164          var tv = timeout
 165          setsockopt(fd, SOL_SOCKET, SO_RCVTIMEO, &tv, socklen_t(MemoryLayout<timeval>.size))
 166  
 167          while true {
 168              let n = recv(fd, buf, bufSize, 0)
 169              if n > 0 {
 170                  data.append(buf, count: n)
 171                  if n < bufSize { break }
 172              } else {
 173                  break
 174              }
 175          }
 176  
 177          guard !data.isEmpty else { return nil }
 178          return try? DIProtocol.decode(data)
 179      }

 198          let cwd = FileManager.default.currentDirectoryPath
 199          let termProgram = ProcessInfo.processInfo.environment["TERM_PROGRAM"] ?? "unknown"
 200          let termSession = currentTermSessionId
 201          let seed = "\(agent)-\(cwd)-\(termProgram)-\(termSession)"

 204          return "\(agent)-\(String(hash, radix: 16))"

 209      static func parseArgs() -> [String: String] {
 210          var result: [String: String] = [:]
 211          let args = CommandLine.arguments
 212          var i = 1
 213          while i < args.count {
 214              let arg = args[i]
 215              if arg.hasPrefix("--"), i + 1 < args.count {
 216                  let key = String(arg.dropFirst(2))
 217                  result[key] = args[i + 1]
 218                  i += 2
 219              } else {
 220                  i += 1
 221              }
 222          }
 223          return result
 224      }

 267      static func buildMessage(agentType: String, hookType: String, sessionId: String, stdinData: [String: Any]?, explicitTool: String? = nil) -> DIMessage {
 268          let hook = hookType.lowercased()
 269  
 270          if hook.contains("pretooluse") || hook.contains("tool_use") || hook == "tooluse" {
 271              let toolName = explicitTool ?? stdinData?["tool_name"] as? String ?? stdinData?["tool"] as? String ?? ""
 272              if isQuestionTool(toolName) {
 273                  return buildQuestionFromPermission(sessionId: sessionId, toolName: toolName, data: stdinData)
 274              }
 275              return buildToolStart(sessionId: sessionId, data: stdinData, explicitTool: explicitTool)
 276          }
 277          if hook.contains("posttooluse") || hook == "patchapply" {
 278              return buildToolComplete(sessionId: sessionId, data: stdinData, explicitTool: explicitTool)
 279          }
 280          if hook.contains("permission") {
 281              return buildPermissionRequest(sessionId: sessionId, data: stdinData)
 282          }
 283          if hook.contains("question") || hook.contains("ask") {
 284              return buildQuestion(sessionId: sessionId, data: stdinData)
 285          }
 286          if hook.contains("plan") {
 287              return buildPlanReview(sessionId: sessionId, data: stdinData)
 288          }
 289          if hook == "subagentstart" || hook == "subagent_start" {
 290              return buildSubagentStart(sessionId: sessionId, agentType: agentType, data: stdinData)
 291          }
 292          if hook == "subagentstop" || hook == "subagentend" || hook == "subagent_stop" || hook == "subagent_end" {
 293              return buildSubagentEnd(sessionId: sessionId, agentType: agentType, data: stdinData)
 294          }
 295          if hook == "precompact" || hook.contains("compact") {
 296              return buildContextCompact(sessionId: sessionId, data: stdinData)
 297          }
 298          if hook == "sessionstart" || hook.contains("session_start") || hook == "userpromptsubmit" {
 299              return buildSessionStart(sessionId: sessionId, agentType: agentType, data: stdinData)
 300          }
 301          if hook == "sessionend" || hook.contains("session_end") || hook == "stop" {
 302              var msg = DIMessage(type: .sessionEnd, sessionId: sessionId)
 303              msg.agentType = agentType
 304              msg.status = stdinData?["last_assistant_message"] as? String
 305                  ?? stdinData?["message"] as? String
 306                  ?? stdinData?["response"] as? String
 307                  ?? stdinData?["result"] as? String
 308              extractTokens(stdinData, into: &msg)
 309              return msg
 310          }
 311  
 312          var msg = buildNotification(sessionId: sessionId, data: stdinData)
 313          msg.agentType = agentType
 314          return msg
 315      }

 317      static func buildSessionStart(sessionId: String, agentType: String, data: [String: Any]?) -> DIMessage {
 318          var msg = DIMessage(type: .sessionStart, sessionId: sessionId)
 319          msg.agentType = agentType
 320          // 空字符串也要走 fallback，否则 ?? 链被截断，TERM_PROGRAM 不会被使用
 321          let termFromData = data?["terminal"] as? String

 323              ?? ProcessInfo.processInfo.environment["TERM_PROGRAM"]
 324              ?? "Terminal"
 325          let tsid = currentTermSessionId
 326          if !tsid.isEmpty { msg.termSessionId = tsid }
 327          let cwdFromData = data?["working_dir"] as? String
 328              ?? data?["cwd"] as? String
 329              ?? data?["projectPath"] as? String
 330              ?? data?["workspace"] as? String
 331              ?? data?["workspaceFolder"] as? String
 332              ?? (data?["workspace_roots"] as? [String])?.first
 333          // 不 fallback 到 FileManager.default.currentDirectoryPath——那是 DIBridge 自身的 cwd，不是用户项目
 334          msg.workingDir = (cwdFromData?.isEmpty == false ? cwdFromData : nil)
 335              ?? ProcessInfo.processInfo.environment["PROJECT_DIR"]
 336          msg.prompt = extractUserPrompt(data)
 337          extractTokens(data, into: &msg)
 338          return msg
 339      }

 341      static func extractUserPrompt(_ data: [String: Any]?) -> String {
 342          guard let raw = data?["prompt"] as? String, !raw.isEmpty else { return "" }
 343          // Codex UserPromptSubmit may include system instructions before the user message.
 344          // If it looks like a system prompt, extract just the last user turn or truncate.
 345          let trimmed = raw.trimmingCharacters(in: .whitespacesAndNewlines)
 346          if trimmed.hasPrefix("You are a") || trimmed.hasPrefix("You will be") || trimmed.hasPrefix("System:") {
 347              // Try to find the actual user message after system instructions
 348              for separator in ["\n\nUser:", "\nUser:", "\n\n> ", "\n---\n"] {
 349                  if let range = trimmed.range(of: separator, options: .backwards) {
 350                      let userPart = trimmed[range.upperBound...].trimmingCharacters(in: .whitespacesAndNewlines)
 351                      if !userPart.isEmpty { return String(userPart) }
 352                  }
 353              }
 354              // Last resort: take the last line if it's short enough to be a user message
 355              if let lastLine = trimmed.split(separator: "\n").last, lastLine.count < 500 {
 356                  return String(lastLine)
 357              }
 358              return ""
 359          }
 360          return raw
 361      }

 391          let toolName = data?["tool_name"] as? String ?? data?["tool"] as? String ?? "unknown"

 408          msg.filePath = findString(in: sources, keys: ["file_path", "path", "filePath"])
 409          return msg

 412      static func descriptionFromToolInput(_ input: [String: Any]?) -> String? {
 413          guard let input else { return nil }
 414          let interestingKeys = ["command", "query", "content", "code", "url", "pattern", "text"]
 415          for key in interestingKeys {
 416              if let val = input[key] as? String, !val.isEmpty {
 417                  return val
 418              }
 419          }
 420          if input.count <= 3 {
 421              let parts = input.compactMap { k, v -> String? in
 422                  guard let s = v as? String, !s.isEmpty else { return nil }
 423                  return "\(k): \(s)"
 424              }
 425              if !parts.isEmpty { return parts.joined(separator: "\n") }
 426          }
 427          return nil
 428      }

 437      static func buildQuestionFromPermission(sessionId: String, toolName: String, data: [String: Any]?) -> DIMessage {
 438          var msg = DIMessage(type: .question, sessionId: sessionId)
 439  
 440          let inputDict = extractToolInput(data)
 441          let questionObj = (inputDict?["questions"] as? [[String: Any]])?.first
 442          let sources: [[String: Any]?] = [questionObj, inputDict, data]
 443  
 444          var header = ""
 445          if let h = questionObj?["header"] as? String, !h.isEmpty { header = h + ": " }
 446          let qText = findString(in: sources, keys: ["question", "text", "message", "description", "prompt"])
 447          msg.questionText = qText.map { header + $0 } ?? toolName
 448  
 449          for src in sources {
 450              guard let src else { continue }
 451              if let opts = src["options"] as? [String] {
 452                  msg.options = opts; break
 453              }
 454              if let opts = src["options"] as? [[String: Any]] {
 455                  msg.options = extractLabels(from: opts); break
 456              }
 457              if let opts = src["choices"] as? [String] {
 458                  msg.options = opts; break
 459              }
 460              if let opts = src["choices"] as? [[String: Any]] {
 461                  msg.options = extractLabels(from: opts); break
 462              }
 463          }
 464  
 465          if let defaultAnswer = findString(in: sources, keys: ["default_answer", "default"]),
 466             msg.options == nil || msg.options!.isEmpty {
 467              msg.options = [defaultAnswer]
 468          }
 469  
 470          return msg
 471      }

 580          answer: String,
 581          stdinData: [String: Any]?,
 582          hookType: String = "PreToolUse"

 619              "hookSpecificOutput": [
 620                  "hookEventName": hookEventName,
 621                  "permissionDecision": "allow",
 622                  "updatedInput": updatedInput
 623              ] as [String: Any]
 624          ]

 626                let str = String(data: data, encoding: .utf8) else {

 422      private func shouldIgnoreMirroredSession(_ message: DIMessage) -> Bool {
 423          guard let agentType = AgentType.from(message.agentType),
 424                agentType == .claudeCode,
 425                message.type == .sessionStart || message.type == .sessionEnd,
 426                let sessionSuffix = mirroredSessionSuffix(from: message.sessionId)
 427          else {
 428              return false
 429          }
 430  
 431          return sessions.contains { session in
 432              isCursorFamily(session.agentType) && suffixCache[session.id] == sessionSuffix
 433          }
 434      }

 437          agentType == .cursor || agentType == .trae

 440      private func mirroredSessionSuffix(from sessionId: String) -> String? {
 441          guard let separator = sessionId.firstIndex(of: "-") else {
 442              return nil
 443          }
 444          let suffix = String(sessionId[sessionId.index(after: separator)...])
 445          return suffix.isEmpty ? nil : suffix
 446      }

 448      private func resolvedAgentType(for message: DIMessage) -> AgentType {
 449          if let explicit = AgentType.from(message.agentType) {
 450              if explicit == .claudeCode, let mirrored = mirroredCursorFamilyAgent(for: message.sessionId) {
 451                  return mirrored
 452              }
 453              return explicit
 454          }
 455  
 456          if let prefix = message.sessionId.split(separator: "-", maxSplits: 1).first,
 457             let inferred = AgentType.from(String(prefix)) {
 458              return inferred
 459          }
 460  
 461          if let mirrored = mirroredCursorFamilyAgent(for: message.sessionId) {
 462              return mirrored
 463          }
 464  
 465          if let terminal = message.terminal, let app = TerminalApp.detect(from: terminal) {
 466              switch app {
 467              case .cursor, .windsurf:
 468                  return .cursor
 469              case .trae, .traeCn:
 470                  return .trae
 471              case .codex:
 472                  return .codex
 473              default:
 474                  break
 475              }
 476          }
 477  
 478          if let existing = sessions.first(where: { $0.id == message.sessionId }) {
 479              return existing.agentType
 480          }
 481  
 482          return .claudeCode
 483      }

 515      func handlePermissionRequest(_ message: DIMessage, respond: @escaping @Sendable (Bool) -> Void) {
 516          if bypassMode {
 517              respond(true)
 518              return
 519          }
 520  
 521          let realAgent = resolvedAgentType(for: message)
 522  
 523          if realAgent == .openCode {
 524              let tool = (message.tool ?? "").trimmingCharacters(in: .whitespacesAndNewlines).lowercased()
 525              let desc = (message.permDescription ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
 526              let path = (message.filePath ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
 527              let diff = (message.diff ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
 528              let isSafeFilesystemPermission = tool == "external_directory"
 529                  || tool == "externaldirectory"
 530              let isPlaceholderPermission = (tool.isEmpty || tool == "unknown")
 531                  && desc.isEmpty
 532                  && path.isEmpty
 533                  && diff.isEmpty
 534              if isPlaceholderPermission || isSafeFilesystemPermission {
 535                  respond(true)
 536                  return
 537              }
 538          }
 539  
 540          let session = findOrCreateSessionForInteraction(message)
 541          session.status = .waitingPermission
 542          session.pendingPermission = PendingPermission(
 543              requestingAgent: realAgent,
 544              tool: message.tool ?? "unknown",
 545              description: message.permDescription ?? "",
 546              diff: message.diff,
 547              filePath: message.filePath,
 548              respond: respond
 549          )
 550          selectedSessionId = session.id
 551          audioEngine?.play(.permissionRequest, session: session)
 552          AppDelegate.shared?.refreshDiagnostics(islandState: diagnosticsIslandState)
 553      }

 555      func handleQuestionRequest(_ message: DIMessage, respond: @escaping @Sendable (String) -> Void, cancel: (@Sendable () -> Void)? = nil) {
 556          if bypassMode, let firstOption = message.options?.first {
 557              respond(firstOption)
 558              return
 559          }
 560  
 561          let realAgent = resolvedAgentType(for: message)
 562          let text = message.questionText ?? ""
 563          let options = message.options ?? []
 564          // Skip stub events (e.g. OpenCode fires a placeholder "Question" with no options
 565          // before the real question arrives). Close the fd so the di-bridge exits cleanly.
 566          let trimmed = text.trimmingCharacters(in: .whitespacesAndNewlines)
 567          guard !trimmed.isEmpty && !(trimmed == "Question" && options.isEmpty) else {
 568              cancel?()
 569              return
 570          }
 571          let session = findOrCreateSessionForInteraction(message)
 572  
 573          // Deduplicate: if this session was answered recently (within 2s) and the
 574          // incoming question text matches (exact or contains — handles [N/M] prefix
 575          // variations from sequential flows), auto-reply with the cached answer so the
 576          // sequential flow can advance to the next question.
 577          if let recent = recentAnswers[session.id],
 578             !options.isEmpty,
 579             Date().timeIntervalSince(recent.timestamp) < 2.0,
 580             (trimmed == recent.questionText
 581              || trimmed.contains(recent.questionText)
 582              || recent.questionText.contains(trimmed)) {
 583              respond(recent.answer)
 584              return
 585          }
 586  
 587          // Close the previous question's fd when superseded by a newer event.
 588          session.pendingQuestion?.cancel?()
 589          session.status = .waitingAnswer
 590          session.pendingQuestion = PendingQuestion(
 591              requestingAgent: realAgent,
 592              text: text,
 593              options: options,
 594              respond: respond,
 595              cancel: cancel
 596          )
 597          selectedSessionId = session.id
 598          audioEngine?.play(.question, session: session)
 599          AppDelegate.shared?.refreshDiagnostics(islandState: diagnosticsIslandState)
 600      }

 701      private func endSession(_ message: DIMessage) {
 702          let agentType = resolvedAgentType(for: message)
 703  
 704          guard let session = sessionForEndMessage(message) else { return }
 705          let alreadyCompleted = session.status == .completed
 706          updateTokenUsage(session: session, message: message)
 707  
 708          if let responseText = message.status, !responseText.isEmpty {
 709              session.agentResponse = responseText
 710              session.statusText = responseText
 711          }
 712  
 713          if !session.agentResponse.isEmpty {
 714              if let lastIdx = session.chatHistory.lastIndex(where: { $0.role == .assistant }) {
 715                  session.chatHistory[lastIdx] = ChatMessage(
 716                      timestamp: Date(), role: .assistant, content: session.agentResponse
 717                  )
 718              } else {
 719                  session.chatHistory.append(
 720                      ChatMessage(timestamp: Date(), role: .assistant, content: session.agentResponse)
 721                  )
 722              }
 723          }
 724  
 725          if !agentType.processNames.isEmpty {
 726              session.status = .idle
 727              session.currentTool = nil
 728              if session.statusText.isEmpty {
 729                  session.statusText = "Done"
 730              }
 731              session.lastActivityTime = Date()
 732          } else {
 733              markCompleted(session)
 734          }
 735          if !alreadyCompleted {
 736              playAssistantReplySoundDebounced(sessionId: session.id)
 737          }
 738  
 739          if session.status == .completed, selectedSessionId == session.id {
 740              selectedSessionId = activeSessions.first?.id
 741          }
 742      }

 754      private func scheduleLingerCleanup() {
 755          guard !hasPendingLingerCleanup else { return }
 756          hasPendingLingerCleanup = true
 757          let linger = completedLingerDuration
 758          DispatchQueue.main.asyncAfter(deadline: .now() + linger + 0.1) { [weak self] in
 759              guard let self else { return }
 760              self.hasPendingLingerCleanup = false
 761              self.visibleSessionsVersion += 1
 762              self.cleanupLingeredSessions()
 763          }
 764      }

 766      func cleanupLingeredSessions() {
 767          // 永久模式（.infinity）不清理任何已完成会话
 768          guard completedLingerDuration != .infinity else {
 769              rebuildSessionIndex()
 770              return
 771          }
 772          // 保留最近的已完成会话，超出上限的按时间淘汰
 773          let completedSessions = sessions
 774              .filter { $0.status == .completed }
 775              .sorted { ($0.completedAt ?? .distantPast) > ($1.completedAt ?? .distantPast) }
 776          if completedSessions.count > 20 {
 777              let toRemove = Set(completedSessions[20...].map(\.id))
 778              sessions.removeAll { toRemove.contains($0.id) }
 779          }
 780          rebuildSessionIndex()
 781          if let selectedSessionId, !sessions.contains(where: { $0.id == selectedSessionId }) {
 782              self.selectedSessionId = activeSessions.first?.id
 783          }
 784          AppDelegate.shared?.refreshDiagnostics(islandState: diagnosticsIslandState)
 785          purgeOrphanedCacheEntries()
 786      }

 973      private func rebuildSessionIndex() {
 974          sessionIndex.removeAll()
 975          suffixCache.removeAll()
 976          subagentMap.removeAll()
 977          for (i, session) in sessions.enumerated() {
 978              sessionIndex[session.id] = i
 979              if let suffix = mirroredSessionSuffix(from: session.id) {
 980                  suffixCache[session.id] = suffix
 981              }
 982              // 构建 subagent 关系
 983              for childId in session.subagentIds {
 984                  if let child = sessions.first(where: { $0.id == childId }) {
 985                      subagentMap[session.id, default: []].append(child)
 986                  }
 987              }
 988          }
 989      }

 995          }

1003      private func sessionById(_ sessionId: String) -> AgentSession? {
1004          if sessionIndex.isEmpty && !sessions.isEmpty { rebuildSessionIndex() }
1005          guard let index = sessionIndex[sessionId] else { return nil }
1006          guard index < sessions.count, sessions[index].id == sessionId else {
1007              rebuildSessionIndex()
1008              guard let freshIndex = sessionIndex[sessionId] else { return nil }
1009              return sessions[freshIndex]
1010          }
1011          return sessions[index]
1012      }

 490      /// Returns false if another instance is already running (exclusive lock held).
 491      private static func acquireSingleInstanceLock() -> Bool {
 492          let dir = (NSHomeDirectory() as NSString).appendingPathComponent(".xisland")
 493          try? FileManager.default.createDirectory(atPath: dir, withIntermediateDirectories: true)
 494          let path = (dir as NSString).appendingPathComponent("instance.lock")
 495          let fd = open(path, O_CREAT | O_RDWR, S_IRUSR | S_IWUSR)
 496          guard fd >= 0 else { return true }
 497          if flock(fd, LOCK_EX | LOCK_NB) != 0 {
 498              close(fd)
 499              return false
 500          }
 501          singleInstanceLockFD = fd
 502          return true
